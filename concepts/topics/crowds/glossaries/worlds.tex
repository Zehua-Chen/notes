\newglossaryentry{triadic-closure}{
  name={triadic closure},
  description={a node satisfies triadic closure property if all of her
  strong ties are strong ties with each other}
}

\newglossaryentry{structural-balance}{
  name={structural balance},
  description={no triangle with 1 or 3 negative edges exist}
}

\newglossaryentry{structural-balance-incomplete-graph}{
  name={structural balance (incomplete graph)},
  description={a graph is balanced iff all edges missing can be fixed so that
  the resulting graph is strongly balanced}
}
