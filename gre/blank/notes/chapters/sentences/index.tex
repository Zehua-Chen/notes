\chapter{句式结构}

\begin{enumerate}
  \item \textbf{\say{to v.}}: 有目标的意思;
  \item \textbf{\say{surprise at}}: not;
  \item \textbf{\say{to A’s credit}}: A值的赞誉;
  \item \textbf{\say{A even B}}: B比A更强;
  \item \textbf{\say{Even adj. A}}: A在adj. 的反面;
  \item \textbf{\say{to as np.(2) np.(1)}}: \say{take np.(1) as np.(2)};
  \item \textbf{\say{so ... as to ...}}: 那么地..., 以至于...;
  \item \textbf{同位语 (\say{x., np.})}: 逗号后面的名词成分解释逗号前的成分;
\end{enumerate}
